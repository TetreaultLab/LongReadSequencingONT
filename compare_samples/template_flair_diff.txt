#!/bin/bash
#SBATCH -N 1 #Number of nodes
#SBATCH -n 8 # number of cores
#SBATCH --mem 32G # memory pool for all cores
#SBATCH -t 2-23:00 # time (D-HH:MM)
#SBATCH -o flair_diff_{0}.%N.%j.log
#SBATCH -e flair_diff_{0}.%N.%j.log
#SBATCH --mail-type=FAIL
#SBATCH --account=rrg-tetreaum
#SBATCH --mail-user={1}

set -euo pipefail

ml apptainer/1.4.5 bedtools/2.31.0

flair_img="/lustre09/project/6019267/shared/tools/splicing/flair/flair_20260320.sif"
intron_prospector_tool="/lustre09/project/6019267/shared/tools/splicing/flair/intronProspector/bin"
project_path="/lustre10/scratch/$USER/{0}/flair_diff"
genome_fa="{2}"
annotation_gtf="{3}"
work_dir="{4}"
manifest_combine="$work_dir/manifest_combine.txt"
manifest_quantify="$work_dir/manifest_quantify.txt"

mkdir -p $project_path

# Clear list
if [[ -f "$project_path/IPjunctions_{0}_list.txt" ]]; then
    rm $project_path/IPjunctions_{0}_list.txt
fi

# Find junctions for each samples
for sample in {5}
do
    bam_file=$(grep -P "^$sample\t" {6} | cut -f3)/alignments/${{sample}}_sorted.bam
    echo -e "\nProcessing $bam_file for junctions"

    apptainer run $flair_img $intron_prospector_tool/intronProspector \
        -g $genome_fa \
        -b $project_path/$sample.IPjunctions.bed \
        -C 0.0 \
        -c $project_path/${{sample}}_intron_calls.tsv \
        $bam_file

    echo -e "$project_path/${{sample}}_intron_calls.tsv" >> $project_path/IPjunctions_{0}_list.txt

done

# Merge junctions into one junctions file
echo -e "\nMerging junction files"
apptainer run $flair_img $intron_prospector_tool/intronProspectorMerge \
    -i $project_path/IPjunctions_{0}_list.txt \
    -b $project_path/IPjunctions_{0}_list_merged.bed

# Generate transcriptomes with FLAIR
for sample in {5}
do
    bam_file=$(grep -P "^$sample\t" {6} | cut -f3)/alignments/${{sample}}_sorted.bam
    echo -e "\nProcessing $bam_file for FLAIR transcriptomes"

    apptainer run $flair_img flair transcriptome \
        -g $genome_fa \
        -b $bam_file \
        -f $annotation_gtf \
        --threads 8 \
        --parallelmode bychrom \
        --junction_support 2 \
        --junction_bed $project_path/IPjunctions_{0}_list_merged.bed \
        -o $project_path/$sample
done

# Combine transcriptomes
echo -e "\nMerging transcriptomes"
apptainer run $flair_img flair combine \
    -m $manifest_combine \
    --include_se \
    -p 0 \
    -f 1 \
    -s \
    -o $project_path/{0}_combined

# Quantify splicing
echo -e "\nRunning FLAIR quantify"
apptainer run $flair_img flair quantify \
    -r $manifest_quantify \
    -i $project_path/{0}_combined.fa \
    --threads 8 \
    --temp_dir $project_path/temp \
    -o $project_path/{0}_quantify \
    --isoform_bed $project_path/{0}_combined.bed \
    --stringent \
    --check_splice \
    --quality 0

# Run differential expression
echo -e "\nRunning FLAIR diff expression"
apptainer run $flair_img flair diffexp \
    -q $project_path/{0}_quantify.counts.tsv \
    --threads 8 \
    --out_dir $project_path/diffexp \
    --out_dir_force

# Run differential splicing
echo -e "\nRunning FLAIR diff splicing"
apptainer run $flair_img flair diffsplice \
    -i $project_path/{0}_combined.bed \
    -q $project_path/{0}_quantify.counts.tsv \
    --threads 8 \
    --test \
    --drim1 3 \
    --drim2 3 \
    --drim3 3 \
    --drim4 3 \
    --conditionA {7} \
    --conditionB {8} \
    --out_dir $project_path/diffsplice \
    --out_dir_force

echo -e "\nFLAIR done!"
