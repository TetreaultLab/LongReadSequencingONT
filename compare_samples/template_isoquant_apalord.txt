#!/bin/sh
#SBATCH -N 1
#SBATCH --cpus-per-task 8
#SBATCH --mem 128G
#SBATCH -t 02-23:59 # time (DD-HH:MM)
#SBATCH -o isoquant_apalord_{0}.%N.%j.log
#SBATCH -e isoquant_apalord_{0}.%N.%j.log
#SBATCH --mail-type=FAIL
#SBATCH --mail-user={1}
#SBATCH --account=rrg-tetreaum

set -euo pipefail

# ---- Environment setup ----
ml gcc r/4.5.0 apptainer python/3.10.13

isoquant="/lustre09/project/6019267/shared/tools/splicing/isoquant"
apalord="/lustre09/project/6019267/shared/tools/polyadenylation/apalord"

export R_LIBS=$apalord/custom_r_4.5.0_lib/ # custom R libary

# ---- Variables -----
isoquant_dir="/lustre10/scratch/$USER/{0}/isoquant"
apalord_dir="/lustre10/scratch/$USER/{0}/apalord"
genome_fa="{2}"
gtf="{3}"

# ---- IsoQuant ----
echo -e "\nRunning IsoQuant\n"
for sample in {4}
do
        bam_file=$(grep -P "^$sample\t" {5} | cut -f3)/alignments/${{sample}}_sorted.bam

        apptainer run $isoquant/isoquant.sif $isoquant/bin/isoquant.py \
                --reference $genome_fa \
                --genedb $isoquant/gencode.v48.primary_assembly.annotation.db \
                --bam $bam_file \
                --data_type nanopore \
                -o $isoquant_dir/$sample
done

# ---- APALORD ----
if [ -d "$apalord_dir" ]; then
        rm -r $apalord_dir
fi
echo -e "\nRunning APALORD\n"
Rscript --vanilla $apalord/apalord.R \
        --case_group {6} \
        --ctrl_group {7} \
        --case_names {8} \
        --ctrl_names {9} \
        -g $gtf \
        -o $apalord_dir \
        -i $isoquant_dir
