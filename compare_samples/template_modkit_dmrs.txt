#!/bin/sh
#SBATCH -N 1
#SBATCH --cpus-per-task 4
#SBATCH --mem 32G
#SBATCH -t 00-01:00 # time (DD-HH:MM)
#SBATCH -o modkit_dmrs_{0}.%N.%j.log
#SBATCH -e modkit_dmrs_{0}.%N.%j.log
#SBATCH --mail-type=FAIL
#SBATCH --mail-user={1}
#SBATCH --account=def-tetreaum

set -euo pipefail

ml htslib/1.22.1

MODKIT="/lustre09/project/6019267/shared/tools/methylation/modkit/dist_modkit_v0.6.1_481e3c9/modkit"
out_dir="/lustre10/scratch/$USER/{0}/modkit_dmrs"
cpg_islands="/lustre09/project/6019267/shared/tools/methylation/modkit/CpG_islands_GRCh38_20260317_cleaned.bed"
ref="{2}"
metadata="{3}"

mkdir -p $out_dir

s=""

# index bedmethyl files
for sample in {4}
do
    condition=$(grep -P "^$sample\t" $metadata | cut -f2)
    pileup=$(grep -P "^$sample\t" $metadata | cut -f3)/results/epi2me/$sample
    
    # If results ar phased
    if [[ -f "$pileup/$sample.wf_mods.1.bedmethyl.gz" ]]; then
        tabix -f -p bed $pileup/$sample.wf_mods.1.bedmethyl.gz
        s+=" -s $pileup/$sample.wf_mods.1.bedmethyl.gz ${{condition}}_${{sample}}"
    fi
    
    if [[ -f "$pileup/$sample.wf_mods.2.bedmethyl.gz" ]]; then
        tabix -f -p bed $pileup/$sample.wf_mods.2.bedmethyl.gz
        s+=" -s $pileup/$sample.wf_mods.2.bedmethyl.gz ${{condition}}_${{sample}}"
    fi

    # If results are not phased
    if [[ -f "$pileup/$sample.wf_mods.bedmethyl.gz" ]]; then
        tabix -f -p bed $pileup/$sample.wf_mods.bedmethyl.gz
        s+=" -s $pileup/$sample.wf_mods.bedmethyl.gz ${{condition}}_$sample"
    fi
done

# Per CpGs
echo -e "\nRunning modkit DMRs for CpGs"
# modkit dmr pair -a <CONTROL_BED_METHYL> -b <EXP_BED_METHYL>
$MODKIT dmr pair \
    {5} \
    -o $out_dir/{0}_cpg.bed \
    --regions-bed $cpg_islands \
    --ref $ref \
    --base C \
    --header \
    --threads 4 \
    --io-threads 4 \
    --force \

# Per base
echo -e "\nRunning modkit DMRs base"
$MODKIT dmr pair \
    {5} \
    -o $out_dir/{0}_base.bed \
    --ref $ref \
    --base C \
    --header \
    --threads 4 \
    --io-threads 4 \
    --force \
