#!/bin/bash
#SBATCH -N 1 #Number of nodes
#SBATCH -n 8 # number of cores
#SBATCH --mem 128G # memory pool for all cores
#SBATCH -t 0-23:00 # time (D-HH:MM)
#SBATCH -o annotate_snps_{0}.%N.%j.log
#SBATCH -e annotate_snps_{0}.%N.%j.log
#SBATCH --mail-type=FAIL
#SBATCH --account=def-tetreaum
#SBATCH --mail-user={1}

set -euo pipefail

ml apptainer/1.4.5

STEPS_DONE="{2}/scripts/steps_done.txt"

output_path="{3}"
VCF="{4}"

mkdir -p $output_path
mkdir -p $SLURM_TMPDIR/{0}

# Running VEP
echo -e "\nRunning VEP\n"
apptainer run /lustre09/project/6019267/shared/tools/variants/annotation/VEP/vep.sif vep \
    --offline \
    --cache \
    --dir_cache /lustre09/project/6019267/shared/tools/variants/annotation/VEP/vep_data/ \
    -i $VCF \
    -o {3}/{0}_vep.vcf \
    --vcf \
    --force_overwrite \
    --stats_file "$output_path/{0}_vep_summary.html" \
    --variant_class \
    --domains \
    --regulatory \
    --canonical \
    --individual_zyg all \
    --hgvs \
    --numbers \
    --pubmed \
    --uniprot \
    --mane \
    --protein \
    --symbol \
    --ccds \
    --biotype \
    --gene_phenotype \
    --mirna \
    --appris \
    --tsl \
    --pick \
    --fork 8

# Running openCravat
echo -e "\nRunning openCravat\n"

env -i PATH=/lustre09/project/6019267/shared/tools/variants/annotation/openCravat_env/bin:/usr/bin:/bin
module --force purge
module load StdEnv/2023 python/3.10
source /lustre09/project/6019267/shared/tools/variants/annotation/openCravat_env/bin/activate

/lustre09/project/6019267/shared/tools/variants/annotation/openCravat_env/bin/oc run \
    {3}/{0}_vep.vcf \
    -a alfa alphamissense cadd clingen clinvar clinvar_acmg dbsnp denovo gerp geuvadis gnomad4 gtex gwas_catalog hg19 interpro mutation_assessor mutationtaster phastcons phylop polyphen2 provean revel sift spliceai thousandgenomes aloft bayesdel clinpred fathmm genehancer hpo mitomap mutpred_indel omim pangolin repeat \
    --mp 8 \
    --debug \
    -l hg38 \
    -d $SLURM_TMPDIR/{0}/ \
    -t text \
    --cleanrun

# Filtering Variants
echo -e "\nFiltering Variants\n"
python -u /lustre09/project/6019267/shared/tools/main_pipelines/long-read/LongReadSequencingONT/filter_variants.py \
    --output $SLURM_TMPDIR/{0}/ \
    --vcf {3}/{0}_vep.vcf \
    --sample {0}

rsync -azh --no-g --no-p $SLURM_TMPDIR/{0}/* {3}/

if [ $? -eq 0 ]; then
    echo "annotate_{0}" >> "$STEPS_DONE"
fi
