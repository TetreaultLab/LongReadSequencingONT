#SBATCH --mem {1}G # memory pool for all cores

module load nextflow
module load apptainer

export NXF_DISABLE_UPDATE_CHECK=true
export NXF_OFFLINE=true
export NXF_SINGULARITY_CACHEDIR="/lustre09/project/6019267/shared/tools/main_pipelines/long-read/epi2me_cache/"
#
WORKDIR="{6}/work"
BAM="{6}/alignments/{4}_sorted.bam"
OUT="{6}/results/{4}"
APPTAINER_CONFIG="/lustre09/project/6019267/shared/tools/main_pipelines/long-read/apptainer_nextflow.config"

    {7} \
    --bam_min_coverage 4 \
    --override_basecaller_cfg {9} \
    -w $WORKDIR \
    --bam $BAM \
    --ref {8} \
    --sample_name {4} \
    --out_dir $OUT \
    --threads {0} \
    --ubam_map_threads {0} \
    --ubam_sort_threads {0} \
    --ubam_bam2fq_threads {0} \
    --modkit_threads {0} \
    -c $APPTAINER_CONFIG \
    -profile apptainer
