#!/bin/bash
#SBATCH -N 1 #Number of nodes
#SBATCH --cpus-per-task 4 # number of cores
#SBATCH --mem 128G # memory pool for all cores
#SBATCH -t 00-11:00 # time (D-HH:MM)
#SBATCH -o longphase_{0}.%N.%j.log
#SBATCH -e longphase_{0}.%N.%j.log
#SBATCH --mail-type=FAIL
#SBATCH --account=rrg-tetreaum
#SBATCH --mail-user={1}

set -euo pipefail

longphase="/lustre09/project/6019267/shared/tools/phasing/longphase/longphase_linux-x64"

WD="/lustre10/scratch/$USER/{2}/results/longphase/{0}"
REF="{3}"
BAM="{4}"
STEPS_DONE="{5}/scripts/steps_done.txt"

mkdir -p $WD

# Extracts per-read methylation tags from the BAM into a VCF for methylation-aware phasing
$longphase modcall \
    -b $BAM \
    -r $REF \
	-s /lustre10/scratch/$USER/{2}/results/epi2me/{0}/{0}.wf_snp.vcf.gz \
	-o $WD/{0}_mod \
    -t 4

# Phases SNP, SV, and methylation VCFs
$longphase phase \
    --ont \
	-b $BAM \
	-r $REF \
	-s /lustre10/scratch/$USER/{2}/results/epi2me/{0}/{0}.wf_snp.vcf.gz \
	--sv-file=/lustre10/scratch/$USER/{2}/results/epi2me/{0}/{0}.wf_sv.vcf.gz \
	--mod-file=$WD/{0}_mod \
	--indels \
	-o $WD/{0}_phased \
	-t 4

# Haplotagging
$longphase haplotag \
    --bam-file $BAM \
	--reference $REF \
	--snp-file $WD/{0}_phased_SNP.vcf \
	--sv-file $WD/{0}_phased_SV.vcf \
	--mod-file $WD/{0}_phased_mod.vcf \
	--out-prefix $WD/{0}_haplotag \
	--cram \
	--threads 4

if [ $? -eq 0 ]; then
    echo "longphase_{0}" >> "$STEPS_DONE"
fi
