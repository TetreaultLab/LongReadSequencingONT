#!/bin/bash
#SBATCH -N 1 #Number of nodes
#SBATCH --cpus-per-task=8 # number of cores
#SBATCH --mem 64G # memory pool for all cores
#SBATCH -t 00-23:00 # time (D-HH:MM)
#SBATCH -o methphaser_{0}.%N.%j.log
#SBATCH -e methphaser_{0}.%N.%j.log
#SBATCH --mail-type=FAIL
#SBATCH --account=rrg-tetreaum
#SBATCH --mail-user={1} # CHANGE TO YOUR EMAIL

set -euo pipefail

source /lustre09/project/6019267/shared/tools/main_pipelines/long-read/launch_pipeline_env/bin/activate
export "PATH=/lustre09/project/6019267/shared/tools/phasing/methphaser-0.0.4/methphaser-0.0.4/:$PATH"
export REF_PATH="/lustre09/project/6019267/shared/tools/references/gencode/GRCh38_p14/GRCh38.primary_assembly.genome.fa"

TOOLDIR="/lustre09/project/6019267/shared/tools/phasing/methphaser-0.0.4/methphaser-0.0.4"
WD="/lustre10/scratch/$USER/{2}/results/methphaser/{0}"
REF="{3}"
E2M="/lustre10/scratch/$USER/{2}/results/epi2me/{0}"
STEPS_DONE="{4}/scripts/steps_done.txt"

mkdir -p $WD
mkdir -p $WD/work

$TOOLDIR/meth_phaser_parallel \
    -b $E2M/{0}.haplotagged.cram \
    -r $REF \
    -g $E2M/{0}.wf_snp.haploblocks.gtf \
    -vc /lustre10/scratch/$USER/{2}/results/hapcut2/{0}/{0}_haplotype.phased.vcf.gz \
    -o $WD \
    -t 8

$TOOLDIR/meth_phaser_post_processing \
    -ib $E2M/{0}.haplotagged.cram \
    -if $WD/work \
    -ov $WD/{0}_methphaser.vcf \
    -ob $WD \
    -vc /lustre10/scratch/$USER/{2}/results/hapcut2/{0}/{0}_haplotype.phased.vcf.gz \
    -t 8

if [ $? -eq 0 ]; then
    echo "methphaser_{0}" >> "$STEPS_DONE"
fi
