#!/bin/bash
#SBATCH -N 1 #Number of nodes
#SBATCH -n 4 # number of cores
#SBATCH --mem 32G # memory pool for all cores
#SBATCH --tmp 1000G # amount of temporary disk space
#SBATCH -t 0-11:00 # time (D-HH:MM)
#SBATCH -o samtools_{0}.%N.%j.log
#SBATCH -e samtools_{0}.%N.%j.log
#SBATCH --mail-type=FAIL
#SBATCH --account=rrg-tetreaum
#SBATCH --mail-user={1}

set -euo pipefail

ml samtools

source /lustre09/project/6019267/shared/tools/main_pipelines/long-read/launch_pipeline_env/bin/activate

STEPS_DONE="{2}/scripts/steps_done.txt"

# Sort
echo "Sort"
samtools sort --threads 3 -m 4G -o $SLURM_TMPDIR/{0}_sorted.bam {3}/{0}.bam 

# Index
echo "Index"
samtools index --threads 3 $SLURM_TMPDIR/{0}_sorted.bam

# Transfer
echo "Transfer"
rsync -avzh --no-g --no-p $SLURM_TMPDIR/{0}_sorted.bam {3}/{0}_sorted.bam
rsync -avzh --no-g --no-p $SLURM_TMPDIR/{0}_sorted.bam.bai {3}/{0}_sorted.bam.bai

echo "Done"
if [ $? -eq 0 ]; then
    echo "samtools_{0}" >> "$STEPS_DONE"
fi
